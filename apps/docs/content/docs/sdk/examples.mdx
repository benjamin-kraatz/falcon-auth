# Integration Examples

## Next.js (App Router)

### 1. Set up the client

```ts title="lib/falcon-auth.ts"
import { createFalconAuthClient } from "@falcon-auth/sdk/client";

export const falconAuth = createFalconAuthClient({
  issuerUrl: process.env.FALCON_AUTH_ISSUER_URL!,
  clientId: process.env.FALCON_AUTH_CLIENT_ID!,
  clientSecret: process.env.FALCON_AUTH_CLIENT_SECRET!,
  redirectUri: `${process.env.NEXT_PUBLIC_APP_URL}/auth/callback`,
  scopes: ["openid", "email", "profile", "offline_access"],
});
```

### 2. Login route

```ts title="app/auth/login/route.ts"
import { falconAuth } from "@/lib/falcon-auth";
import { cookies } from "next/headers";
import { redirect } from "next/navigation";

export async function GET() {
  const { url, state, codeVerifier } = await falconAuth.getAuthorizationUrl();

  const cookieStore = await cookies();
  cookieStore.set("falcon_state", state, {
    httpOnly: true,
    secure: process.env.NODE_ENV === "production",
    sameSite: "lax",
    maxAge: 600,
    path: "/",
  });
  cookieStore.set("falcon_code_verifier", codeVerifier, {
    httpOnly: true,
    secure: process.env.NODE_ENV === "production",
    sameSite: "lax",
    maxAge: 600,
    path: "/",
  });

  redirect(url);
}
```

### 3. Callback route

```ts title="app/auth/callback/route.ts"
import { falconAuth } from "@/lib/falcon-auth";
import { cookies } from "next/headers";
import { NextResponse } from "next/server";

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const code = searchParams.get("code");
  const state = searchParams.get("state");
  const error = searchParams.get("error");

  if (error) {
    return NextResponse.redirect(new URL("/?error=auth_failed", request.url));
  }

  const cookieStore = await cookies();
  const storedState = cookieStore.get("falcon_state")?.value;
  const codeVerifier = cookieStore.get("falcon_code_verifier")?.value;

  if (!state || state !== storedState || !code || !codeVerifier) {
    return new Response("Invalid OAuth state", { status: 400 });
  }

  const tokens = await falconAuth.exchangeCode({ code, codeVerifier });
  const user = await falconAuth.getUserInfo(tokens.accessToken);

  // Create your session here (e.g., iron-session, next-auth, etc.)
  // Example: store user in cookie
  const response = NextResponse.redirect(new URL("/dashboard", request.url));
  response.cookies.set("user_session", JSON.stringify({
    userId: user.sub,
    name: user.name,
    email: user.email
  }), {
    httpOnly: true,
    secure: process.env.NODE_ENV === "production",
    maxAge: 60 * 60 * 24 * 7, // 1 week
  });

  // Clean up PKCE cookies
  response.cookies.delete("falcon_state");
  response.cookies.delete("falcon_code_verifier");

  return response;
}
```

## Express / Hono

```ts title="src/routes/auth.ts"
import { createFalconAuthClient } from "@falcon-auth/sdk/client";
import { Hono } from "hono";
import { getCookie, setCookie, deleteCookie } from "hono/cookie";

const auth = new Hono();

const falconAuth = createFalconAuthClient({
  issuerUrl: process.env.FALCON_AUTH_ISSUER_URL!,
  clientId: process.env.FALCON_AUTH_CLIENT_ID!,
  clientSecret: process.env.FALCON_AUTH_CLIENT_SECRET!,
  redirectUri: `${process.env.APP_URL}/auth/callback`,
});

auth.get("/login", async (c) => {
  const { url, state, codeVerifier } = await falconAuth.getAuthorizationUrl();

  setCookie(c, "falcon_state", state, { httpOnly: true, secure: true, maxAge: 600 });
  setCookie(c, "falcon_code_verifier", codeVerifier, { httpOnly: true, secure: true, maxAge: 600 });

  return c.redirect(url);
});

auth.get("/callback", async (c) => {
  const code = c.req.query("code");
  const state = c.req.query("state");

  const storedState = getCookie(c, "falcon_state");
  const codeVerifier = getCookie(c, "falcon_code_verifier");

  if (state !== storedState || !code || !codeVerifier) {
    return c.text("Invalid state", 400);
  }

  const tokens = await falconAuth.exchangeCode({ code, codeVerifier });
  const user = await falconAuth.getUserInfo(tokens.accessToken);

  // Create session...
  deleteCookie(c, "falcon_state");
  deleteCookie(c, "falcon_code_verifier");

  return c.redirect("/dashboard");
});

export default auth;
```

## Machine-to-Machine (Client Credentials)

For server-to-server API calls:

```ts
import { createFalconAuthClient } from "@falcon-auth/sdk/client";

const auth = createFalconAuthClient({
  issuerUrl: "https://auth.falcon.example.com",
  clientId: "your-service-client-id",
  clientSecret: "your-service-client-secret",
  redirectUri: "", // Not needed for client_credentials
});

// Request a token using client credentials
const response = await fetch(
  "https://auth.falcon.example.com/api/auth/oauth2/token",
  {
    method: "POST",
    headers: { "Content-Type": "application/x-www-form-urlencoded" },
    body: new URLSearchParams({
      grant_type: "client_credentials",
      client_id: "your-service-client-id",
      client_secret: "your-service-client-secret",
      scope: "openid",
    }),
  }
);

const { access_token } = await response.json();
// Use access_token to call protected FALCON APIs
```
